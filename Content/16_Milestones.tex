\section{Milestones and Phases}
\minibib

\subsection{Defense Acquisition System Overview}
The\ac{das} is the governance framework that aligns requirements, resourcing, and acquisition execution to deliver warfighter capability. It is policy-driven by DoDD~5000.01 and process-enabled through DoDI~5000.02, using the\ac{aaf} to tailor pathways while preserving statutory oversight and Navy-unique controls~\autocite{DoDD5000-01,DoDI5000-02}.

\subsection{Adaptive Acquisition Framework Pathways (Board Snapshot)}
\begin{description}
	\item[\textbf{\ac{mca}.}] Default for complex weapon systems; uses phased milestones with mandatory statutory certifications; emphasizes affordability trades, test rigor, and Navy two-pass / seven-gate touchpoints~\autocite{DoDI5000-85}.
	\item[\textbf{\ac{mta}.}] Rapid prototyping or rapid fielding to deliver an operational capability within five years; minimizes documentation, but demands robust transition planning to\ac{mca} or sustainment funding~\autocite{DoDI5000-80}.
	\item[\textbf{\ac{ucaacq}.}] Accelerated path to field urgent capability within 2~years or provide an interim solution in less than 5; leverages tailored test, streamlined acquisition decision memoranda, and dedicated senior oversight~\autocite{DoDI5000-81}.
	\item[\textbf{Software Acquisition Pathway.}] Iterative, DevSecOps-focused delivery of software increments with continuous integration, enterprise cloud alignment, and recurring user feedback cadences~\autocite{DoDI5000-87}.
	\item[\textbf{Acquisition of Services Pathway.}] Structures service requirements, performance metrics, and contracting strategies to assure mission outcomes and oversight thresholds (e.g., Multi-Functional\ac{ipt} requirements review)\autocite{DoDI5000-74}.
	\item[\textbf{Defense Business Systems Pathway.}] Rationalizes business processes, data standards, and\ac{bcac} to unlock enterprise interoperability before investing in solutions~\autocite{DoDI5000-75}.
\end{description}
\note{Pathways are not exclusive---an\ac{edo} may begin in\ac{mta} for prototyping, then transition to\ac{mca} for production once requirements and sustainment plans stabilize. Programs can also leverage\ac{ucaacq} for an initial bridge while planning a transition to\ac{mca}.}

\subsection{Urgent Operational Needs Taxonomy}
\begin{description}
	\item[\textbf{\ac{juon}.}] Joint urgent gap requiring Combatant Commander sponsorship; validated by the Joint Rapid Acquisition Cell with rapid fielding oversight~\autocite{CJCSI3470-01H}.
	\item[\textbf{\ac{jeon}.}] Joint emergent gap tied to potential contingency; enables deliberate analysis but retains joint prioritization and Board-level visibility~\autocite{CJCSI5123-01H}.
	\item[\textbf{\ac{uon}.}] Service-specific urgent need (e.g., Fleet Commanders); informs Navy resource prioritization and can transition to\ac{juon}/\ac{jeon} if joint impact grows~\autocite{OPNAVINST5000-42E}.
\end{description}

\subsection{Baseline Discipline: Acquisition Program Baselines, Deviations, and Nunn--McCurdy}
\begin{description}
	\item[\textbf{\ac{apb}.}] The cost-schedule-performance contract between the\ac{mdauthority}, resource sponsor, and\ac{peo}; defines objective and threshold values that bound trade space~\autocite{DoDI5000-85}.
	\item[\textbf{Program Deviations.}] Triggered when performance, schedule, or cost thresholds breach; the Program Manager must notify the\ac{mdauthority} within 30~days and submit a corrective action plan within 60~\autocite{DoDI5000-85}.
	\item[\textbf{Nunn--McCurdy Act.}] Statutory reporting when\ac{pauc} or\ac{apuc} breaches significant ($\geq$15\% current / 30\% original) or critical ($\geq$25\% current / 50\% original) thresholds; critical breaches require certification to Congress within 60~days or program termination~\autocite{USC-4371-NunnMcCurdy}.
\end{description}

\subsection{JCIDS Flow into the Defense Acquisition System}
\begin{description}
	\item[\textbf{\ac{icd}.}] Supports the\ac{mdd} with validated capability needs and initial attributes~\autocite{CJCSI5123-01H}.
	\item[\textbf{\ac{cdd}.}] Provides\acp{kpp} and\acp{ksa} to anchor Milestone~B and the Technical Requirements Baseline~\autocite{CJCSI5123-01H}.
	\item[\textbf{\ac{cpd}.}] Updates\ac{kpp}/\ac{ksa} values for production decisions and sustainment planning at Milestone~C~\autocite{CJCSI5123-01H}.
\end{description}
\note{Maintain requirements traceability matrices linking\ac{kpp} thresholds to test plans and\ac{apb} values for board discussions.}

\subsection{Acquisition Strategy Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Documents tailored acquisition approach, contract strategy, intellectual property posture, and affordability constraints for\ac{mdauthority} approval~\autocite{DoDI5000-85}.
	\item[\textbf{Key Contents.}] Pathway selection, phasing plan, incentive structure, test and evaluation strategy, risk burn-down, sustainment concept, cybersecurity, international cooperation, and business case for software/data management~\autocite{DoDI5000-85,DoDD5000-01}.
	\item[\textbf{Approval Event.}]\ac{mdauthority} concurrence at Milestone~A (initial) with updates prior to Milestone~B and significant re-baselines; Navy 2-pass/7-gate integrates resource validation before signature~\autocite{OPNAVINST5000-42E}.
\end{description}

\subsection{International Acquisition Objectives}
\begin{description}
	\item[\textbf{Foreign Military Sales (FMS).}] Strengthen partner capability while leveraging U.S. economies of scale; requires disclosure review and Letter of Offer and Acceptance alignment~\autocite{DoDI5000-85}.
	\item[\textbf{Direct Commercial Sales (DCS).}] Contractor-led exports; Program Office ensures configuration control and technology-security compliance~\autocite{DoDI5000-85}.
	\item[\textbf{Cooperative Development.}] Shared investment with allies (e.g., Memoranda of Understanding) to reduce life-cycle cost and enhance interoperability~\autocite{DoDI5000-85}.
	\item[\textbf{Armaments Cooperation / International Programs Security.}] Integrates disclosure, export licensing, and technology protection plans into the Acquisition Strategy~\autocite{DoDI5000-85}.
\end{description}

\subsection{Navy Two-Pass / Seven-Gate Review}
\begin{description}
	\item[\textbf{Pass 1 -- Capability.}] Led by OPNAV~N9 to mature the requirement: Gate~1 validates the capability gap, Gate~2 refines solutions and mission analysis, Gate~3 assesses affordability and alignment with Navy enterprises, and Gate~4 authorizes program initiation (supports\ac{mdd})\autocite{OPNAVINST5000-42E}.
	\item[\textbf{Pass 2 -- Resourcing.}] Led by OPNAV~N8 to align budgeting: Gate~5 verifies contracting/test readiness before Milestone~B (often synchronized with the Development RFP Release Decision), Gate~6 confirms production and sustainment resources prior to Milestone~C, and Gate~7 evaluates sustainment performance and modernization plans~\autocite{OPNAVINST5000-42E}.
\end{description}
\note{For board prep, highlight where SECNAV Gate Reviews or Defense Acquisition Board touchpoints overlay Navy gates to demonstrate cross-tier synchronization.}

\subsection{Major Capability Acquisition Lifecycle Overview}
\begin{description}
	\item[\textbf{\ac{msa}.}] Shape the trade space and select a preferred materiel solution; conclude with an\ac{aoa} recommendation and technology maturation plan~\autocite{DoDI5000-85}.
	\item[\textbf{\ac{tmrr}.}] Mature technologies, refine requirements, and reduce integration risk to enter\ac{emd}; manage competitive prototyping where feasible~\autocite{DoDI5000-85}.
	\item[\textbf{\ac{emd}.}] Complete detailed design, verify system performance, and prepare for production and deployment decisions~\autocite{DoDI5000-85}.
	\item[\textbf{\ac{pd.phase}.}] Produce capability, conduct\ac{iote}, field initial units, and decide on\ac{frp}~\autocite{DoDI5000-85}.
	\item[\textbf{\ac{os.phase}.}] Sustain readiness, execute modernization, and plan for disposal while feeding affordability back into\ac{ppbe} cycles~\autocite{DoDI5000-85}.
\end{description}

\Fig{\ref{fig:mca_schedule_placeholder}} Shows the major phases, review and decision points within\acp{mca}. The following are the reviews that occur during the phases:
\begin{description}
  \item[\textbf{\ac{asr}.}] Validates the preferred materiel solution and technology maturation approach emerging from the\ac{aoa}; inputs include threat assessments, cost-schedule trades, draft system concepts, and updated Technology Development Strategy /\ac{sep} outlines, and the primary output is an\ac{asr} report that authorizes drafting authoritative system-level requirements for the upcoming\ac{srr}~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{srr}.}] Confirms that system-level performance, interface, cyber, and logistics requirements trace to the validated\ac{icd} /\ac{cdd} and are testable; requires a draft System Specification, mission threads, interface control documentation, and technical performance measures, and produces a baselined system requirements set and updated verification strategy feeding\ac{sfr} planning~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{sfr}.}] Demonstrates that the functional architecture and allocated requirements can satisfy the mission needs within cost/schedule constraints; leverages functional allocation matrices, preliminary hazard analyses, and draft reliability growth plans, and yields an approved functional baseline plus a risk/opportunity burn-down plan ahead of\ac{pdr}~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{pdr}.}] Assesses whether the allocated baseline and preliminary design (drawings, models, manufacturability assessments, support concepts) are mature enough to enter detailed design; the review outputs include an approved allocated baseline, configuration-controlled action list, and direction for completing qualification hardware prior to\ac{cdr}~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{cdr}.}] Confirms the detailed design meets all requirements with acceptable technical risk; inputs include product baseline technical data packages, qualification/analysis results, producibility assessments, and cybersecurity artifacts, and the decision produces an approved product baseline plus direction to release build-to packages and procure long-lead items~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{trr}.}] Evaluates readiness to execute a specific test event by reviewing test articles, instrumentation, facilities, procedures, safety releases, and deficiency dispositions; a successful\ac{trr} outputs an authorization to start test with documented entrance/exit criteria and risk mitigations~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{svr}.}] Verifies the integrated system meets the product baseline and is ready for production decision reviews; it uses completed verification cross-reference matrices, configuration audits, and qualification /\ac{dt} results to produce a verified system baseline, deficiency adjudication list, and recommendation for\ac{frpdr} preparation~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{otrr}.}] Certifies that production-representative assets, training, logistics products, safety artifacts, and test ranges are ready to support initial operational test; the review outputs the operational test readiness certification and any restrictions or corrective actions before\ac{dote} concurrence~\autocite{DoDI5000-85,DoDAcqGuidebook}.
  \item[\textbf{\ac{prr}.}] Determines whether manufacturing processes, supply chain, quality systems, and sustainment enablers can support\ac{lrip} /\ac{frp} quantities; requires control plans, process capability data, staffing/budget burn rates, and yields a decision to proceed into production along with corrective actions for any shortfalls~\autocite{DoDI5000-85,DoDAcqGuidebook}.
\end{description}

\subsection{Materiel Development Decision}
\begin{description}
	\item[\textbf{Inputs.}] Validated\ac{icd}, initial affordability analysis, preliminary acquisition strategy, and resource sponsor endorsement~\autocite{DoDI5000-02}.
	\item[\textbf{Decision.}]\ac{mdauthority} authorizes entry into\ac{msa} (or an alternate pathway) and designates the lead\ac{peo}/\ac{pm}~\autocite{DoDI5000-85}.
	\item[\textbf{Outputs.}]\ac{adm}, competitive\ac{aoa} guidance, and direction to form the\ac{ipt} structure~\autocite{DoDI5000-85}.
\end{description}

\subsection{Materiel Solution Analysis Phase Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Analyze alternatives, define the preferred materiel solution, and scope technology maturity and risk reduction tasks~\autocite{DoDI5000-85}.
	\item[\textbf{Key Activities.}] Conduct\ac{aoa}, establish initial system performance metrics, develop early test strategies, and plan technology maturation for critical subsystems~\autocite{DoDI5000-85}.
	\item[\textbf{Entrance Criteria.}] Approved\ac{mdd}, resourced\ac{aoa} study guidance, and draft acquisition strategy~\autocite{DoDI5000-02}.
	\item[\textbf{Exit Criteria.}] Approved\ac{tmrr} acquisition strategy update, technology risk assessments (i.e.,\ac{trl}~$\geq$~3--4), and draft\ac{cdd} inputs for Joint staffing~\autocite{DoDI5000-85}.
\end{description}

\subsection{Milestone~A Focus Areas}
\begin{description}
	\item[\textbf{Purpose.}] Authorize entry into\ac{tmrr} and approve the initial Acquisition Strategy and\ac{apb}~\autocite{DoDI5000-85}.
	\item[\textbf{Board Topics.}]\ac{aoa} results, technology maturation plans, affordability caps, cost estimates, and early test strategies~\autocite{DoDI5000-85}.
	\item[\textbf{Outputs.}]\ac{adm}, approved initial\ac{apb}, direction for competitive prototyping, and signed Technology Development Strategy if required~\autocite{DoDI5000-85}.
\end{description}

\subsection{Technology Maturation and Risk Reduction Phase Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Reduce integration risk, mature the Technical Baseline, and prepare for Engineering \& Manufacturing Development source selection~\autocite{DoDI5000-85}.
	\item[\textbf{Key Activities.}] Competitive prototyping (if directed), systems engineering trade studies, reliability growth, cyber-resilience planning, and\ac{cddv}~\autocite{DoDI5000-85}.
	\item[\textbf{Entrance Criteria.}] Approved Milestone~A\ac{adm}, funded risk-reduction plan, and established\ac{ims} aligned to Navy Gate~5 timeline~\autocite{DoDI5000-85,OPNAVINST5000-42E}.
	\item[\textbf{Exit Criteria.}] Technology Readiness Assessments showing\ac{trl}~$\geq$~6, draft Milestone~B documentation (e.g.,\ac{sep},\ac{temp}), and configured product baseline ready for the\ac{drfprd}~\autocite{DoDI5000-85}.
\end{description}

\subsection{Milestone~B Focus Areas}
\begin{description}
	\item[\textbf{Purpose.}] Authorize entry into\ac{emd} and establish the program as a major defense acquisition program with formal reporting~\autocite{DoDI5000-85}.
	\item[\textbf{Board Topics.}] Affordability caps, independent cost estimates,\ac{temp} approval,\ac{lrip} planning, and production-representative prototype evidence~\autocite{DoDI5000-85}.
	\item[\textbf{Outputs.}] Updated\ac{apb}, approved Acquisition Strategy revision, and contractual authority to release\ac{emd} \glspl{rfp}~\autocite{DoDI5000-85}.
\end{description}

\subsection{Engineering and Manufacturing Development Phase Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Complete detailed design, build and test engineering development models, and validate manufacturing processes~\autocite{DoDI5000-85}.
	\item[\textbf{Key Activities.}]\ac{cdr}, system verification, developmental test events, cybersecurity accreditations, and sustainment baseline development~\autocite{DoDI5000-85}.
	\item[\textbf{Entrance Criteria.}] Approved Milestone~B\ac{adm} with resourced contracts, finalized technical baseline, and product-support planning aligned to Navy Gate~6~\autocite{DoDI5000-85}.
	\item[\textbf{Exit Criteria.}] Successful\ac{iote} readiness, manufacturing readiness $\geq$~\ac{trl}~8, and validated sustainment strategies supporting Milestone~C~\autocite{DoDI5000-85}.
\end{description}

\subsection{Milestone~C Focus Areas}
\begin{description}
	\item[\textbf{Purpose.}] Authorize production and deployment following demonstration of operational effectiveness and suitability~\autocite{DoDI5000-85}.
	\item[\textbf{Board Topics.}]\ac{iote} results, sustainment and supply-chain readiness, affordability assessments, and updated manpower/training plans~\autocite{DoDI5000-85}.
	\item[\textbf{Outputs.}]\ac{frpdr} plan, updated\ac{apb}, and authority to obligate procurement funding for production lots~\autocite{DoDI5000-85}.
\end{description}

\subsection{Production and Deployment Phase Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Produce and deliver the system while validating operational performance and support concepts~\autocite{DoDI5000-85}.
	\item[\textbf{Key Activities.}]\ac{lrip} execution,\ac{iote},\ac{csb} governance, and ramp to Full-Rate Production~\autocite{DoDI5000-85}.
\item[\textbf{Exit Criteria.}]\ac{frp} approval, attainment of\ac{ioc}, and sustainment plans ready for fleet release~\autocite{DoDI5000-85}.
\end{description}

\subsection{Operations and Support Phase Essentials}
\begin{description}
	\item[\textbf{Purpose.}] Sustain mission effectiveness, execute modernization, and manage total ownership cost until disposal~\autocite{DoDI5000-85}.
	\item[\textbf{Key Activities.}] Performance-Based Logistics execution, Depot maintenance planning, Post-Deployment reviews, and\acp{csb} for major updates~\autocite{DoDI5000-85}.
	\item[\textbf{Milestones.}]\ac{foc}, disposal decision documentation, and updates to capability requirements feeding future planning cycles~\autocite{DoDI5000-85}.
\end{description}
