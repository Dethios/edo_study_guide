\section{Combatant Commands (Reference)}
\noindent\textbf{\ac{aor}:} The geographic (or astrographic) area assigned to a \ac{ccdr} for missions and forces. (Title~10) % cite earlier laws in main refs if desired

\textbf{Geographic (7):}
\begin{itemize}
	\item USAFRICOM
	\item USCENTCOM
	\item USEUCOM
	\item USINDOPACOM
	\item USNORTHCOM
	\item USSOUTHCOM
	\item USSPACECOM\footnote{USSPACECOM is treated by DoD as a \textbf{geographic (astrographic)} command with an AOR beginning at the K\'{a}rm\'{a}n line ($\sim$100 km)}\\
\end{itemize}
\textbf{Functional (4):}
\begin{itemize}
	\item USSOCOM
	\item USSTRATCOM
	\item USTRANSCOM
	\item USCYBERCOM
\end{itemize}
