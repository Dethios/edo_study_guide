\subsection{HQ Codes (must-know one-liners)}

\begin{description}
	\item[\textbf{SEA 01 -- Comptroller.}]\ac{bso}; leads\ac{navsea} financial governance.
	\item[\textbf{SEA 02 -- Contracts.}] Contracting authority and policy.
	\item[\textbf{SEA 03 -- Cyber Engineering and Digital.}] Enterprise cyber engineering and digital transformation.
	\item[\textbf{SEA 04 -- Industrial Operations.}] Shipyards, logistics, and quality-assurance oversight.
	\item[\textbf{SEA 05 -- Chief Engineer (\gls{ta}).}] Sets engineering standards and certifies designs.
	\item[\textbf{SEA 06 -- Sustainment.}] Life-cycle product support.
	\item[\textbf{SEA 07 -- Undersea Warfare.}] Dual-hatted as\ac{peouws}; provides life-cycle support for in-service submarine and undersea forces.
	\item[\textbf{SEA 08 -- Nuclear Propulsion.}] Manages all technical matters pertaining to naval reactors. Triple-hat duties:
      \begin{description}
        \item[\textbf{SEA 08 Deputy Commander.}] Leads nuclear propulsion activities within\ac{navsea}.
        \item[\textbf{Director, Naval Nuclear Propulsion (\ac{opnav}~00N).}] Serves as the Navy's principal nuclear propulsion authority.
        \item[\textbf{Deputy Administrator for Naval Reactors.}] Interfaces with the National Nuclear Security Administration and the Department of Energy.
      \end{description}
	\item[\textbf{SEA 09 -- Safety and Regulatory Compliance.}] Strengthens and aligns safety oversight and reporting.
	\item[\textbf{SEA 10 -- Total Force and Corporate Operations.}] Manages the workforce, corporate services, and governance.
	\item[\textbf{SEA 21 -- In-Service Ships (dual-hat\ac{cnrmc}).}] Oversees fleet surface-ship sustainment and modernization.\acp{pmo} under SEA 21:
      \begin{description}
        \item[\textbf{PMS 321.}] Unmanned small combatants and amphibious ships.
        \item[\textbf{PMS 326.}] International fleet support.
        \item[\textbf{PMS 339.}] Surface training systems.
        \item[\textbf{PMS 421.}] Large surface combatant modernization and sustainment.
        \item[\textbf{PMS 443.}] Bridge integration and HM\&E sustainment.
        \item[\textbf{PMS 451.}] Destroyer Modernization~2.0 portfolio.
        \item[\textbf{SEA 21I.}] Inactive Ships Directorate.
        \item[\textbf{\ac{surfmepp} Activity.}] Surface Maintenance Engineering Planning Program.
      \end{description}
\end{description}
